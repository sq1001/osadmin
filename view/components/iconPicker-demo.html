图标选择器示例

本页面演示 IconPicker 图标选择器组件的使用方法，支持 fontClass 和 unicode 两种模式

FontClass 模式
选中的图标： 未选择
Unicode 模式
选中的图标： 未选择
自定义配置
带回调函数
回调信息：
禁用搜索
禁用分页

使用示例

// 基本用法 - FontClass 模式
layui.use(['iconPickerMod'], function() {
  var iconPicker = layui.iconPickerMod;
  
  iconPicker.render({
    elem: '#iconInput',
    type: 'fontClass',
    click: function(data) {
      console.log('选中的图标:', data.icon);
    }
  });
});

// Unicode 模式
iconPicker.render({
  elem: '#iconInput',
  type: 'unicode',
  click: function(data) {
    console.log('选中的图标:', data.icon);
  }
});

// 自定义配置
iconPicker.render({
  elem: '#iconInput',
  type: 'fontClass',
  page: true,
  limit: 12,
  cellWidth: '50px',
  search: true,
  click: function(data) {
    console.log('图标:', data.icon);
  },
  success: function(d) {
    console.log('渲染成功:', d);
  }
});

// 手动选中图标
iconPicker.checkIcon('filterName', 'layui-icon-home');
