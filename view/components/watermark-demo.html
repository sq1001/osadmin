指定容器水印

这是一个指定容器的水印示例区域

水印只会显示在这个容器内，不会影响页面其他部分

适用于需要局部水印的场景，如：文档预览区、图片展示区等

动态更新

使用示例

// 基本用法
layui.use(['watermarkMod'], function() {
  var Watermark = layui.watermarkMod;
  
  var wm = new Watermark({
    content: '机密文档'
  });
});

// 指定容器水印
var wm = new Watermark({
  content: '内部资料',
  appendTo: '#document-area',  // 指定容器
  opacity: 0.15,
  rotate: -20
});

// 完整配置
var wm = new Watermark({
  content: '水印文本',
  appendTo: 'body',
  width: 150,
  height: 20,
  rowSpacing: 60,
  colSpacing: 30,
  rotate: -15,
  opacity: 0.1,
  fontSize: 14,
  fontColor: '#000000',
  fontFamily: 'Microsoft YaHei, sans-serif',
  zIndex: 999999
});

// 更新水印文本
wm.upload('新的水印内容');

// 重新渲染
wm.render({
  content: '更新后的水印',
  opacity: 0.2
});

// 销毁水印
wm.destroy();
