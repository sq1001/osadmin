Toast 提示示例

本页面演示 Toast 组件的使用方法，支持多种类型、位置、显示模式、消散动画等自定义配置

消息类型
显示位置
显示模式 新
堆叠模式：多条消息同时显示；替换模式：新消息替换旧消息
数量限制 新
超出数量限制时，最旧的消息会以消散动画移除
消散动画 新
消散动画：淡出 + 缩小 + 高度收缩，形成消散效果
持续时间
加载状态
自定义配置
批量操作
回调函数

使用示例

// 基本用法
toast.success('操作成功！');
toast.error('操作失败！');
toast.warning('警告信息');
toast.info('提示信息');

// 显示位置
toast.show('内容', { offset: 'rt' });  // 右上角
toast.show('内容', { offset: 'rb' });  // 右下角
toast.show('内容', { offset: 'lt' });  // 左上角
toast.show('内容', { offset: 'lb' });  // 左下角
toast.show('内容', { offset: 'tc' });  // 顶部居中
toast.show('内容', { offset: 'bc' });  // 底部居中
toast.show('内容', { offset: 'lc' });  // 左侧居中
toast.show('内容', { offset: 'rc' });  // 右侧居中

// 显示模式（新功能）
toast.show('消息', { mode: 'stack' });    // 堆叠模式（默认）
toast.show('消息', { mode: 'replace' });  // 替换模式

// 数量限制（新功能）
toast.show('消息', { maxCount: 3 });      // 最多显示 3 条
toast.show('消息', { maxCount: 0 });      // 无限制

// 消散动画（新功能）
toast.show('消息', { 
  dismissEffect: true,      // 启用消散动画（默认）
  dismissDuration: 400      // 消散动画时长
});
// 自定义样式类（可选左边框样式）
toast.success('消息', { className: 'toast-success' });  // 绿色左边框
toast.error('消息', { className: 'toast-error' });      // 红色左边框
toast.warning('消息', { className: 'toast-warning' });  // 橙色左边框
toast.info('消息', { className: 'toast-info' });        // 蓝色左边框

// 禁用消散动画
toast.show('消息', { dismissEffect: false });

// 自定义配置
toast.show('自定义内容', {
  title: '自定义标题',
  icon: 'success',
  time: 3000,
  offset: 'rt',
  mode: 'stack',
  maxCount: 5,
  dismissEffect: true,
  dismissDuration: 400,
  closable: true,
  success: function(id, elem) {
    console.log('Toast已显示:', id);
  },
  cancel: function(id, elem) {
    console.log('Toast即将关闭:', id);
    return true;
  }
});

// 加载状态
var loadId = toast.loading('加载中...');
setTimeout(function() {
  toast.dismiss(loadId);  // 使用 dismiss 触发消散动画
}, 2000);

// 关闭所有
toast.clear();        // 立即关闭所有
toast.dismissAll();   // 消散所有（带动画）

配置项说明

配置项 类型 默认值 说明
title String '提示' Toast 标题
icon String '' 图标类型：success/error/warning/info/loading 或自定义 HTML
time Number 3000 自动关闭时间，0 表示不自动关闭
offset String 'rt' 显示位置：rt/rb/lt/lb/tc/bc/lc/rc
mode String 'stack' 显示模式：stack(堆叠) / replace(替换)
maxCount Number 5 最大显示数量，0 表示无限制
dismissEffect Boolean true 是否启用消散动画
dismissDuration Number 400 消散动画时长
closable Boolean true 是否显示关闭按钮
className String '' 自定义 CSS 类名
success Function - 显示成功回调，参数
cancel Function - 关闭前回调，返回 false 可阻止关闭
