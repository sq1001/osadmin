Toast 提示示例

本页面演示 Toast 组件的使用方法，支持多种类型、位置和自定义配置

消息类型
显示位置
持续时间
加载状态
自定义配置
多条消息
回调函数

使用示例

// 基本用法
toast.success('操作成功！');
toast.error('操作失败！');
toast.warning('警告信息');
toast.info('提示信息');

// 显示位置
toast.show('内容', { offset: 'rt' });  // 右上角
toast.show('内容', { offset: 'rb' });  // 右下角
toast.show('内容', { offset: 'lt' });  // 左上角
toast.show('内容', { offset: 'lb' });  // 左下角
toast.show('内容', { offset: 'tc' });  // 顶部居中
toast.show('内容', { offset: 'bc' });  // 底部居中
toast.show('内容', { offset: 'lc' });  // 左侧居中
toast.show('内容', { offset: 'rc' });  // 右侧居中

// 自定义配置
toast.show('自定义内容', {
  title: '自定义标题',
  icon: 'success',
  time: 3000,
  offset: 'rt',
  closable: true,
  success: function(id, elem) {
    console.log('Toast已显示:', id);
  },
  cancel: function(id, elem) {
    console.log('Toast即将关闭:', id);
    return true;
  }
});

// 加载状态
var loadId = toast.loading('加载中...');
setTimeout(function() {
  toast.close(loadId);
}, 2000);

// 关闭所有
toast.clear();
